from sklearn.datasets import load_iris
import pandas as pd

iris = load_iris()
df = pd.DataFrame(iris.data, columns=iris.feature_names)

get_target_names = lambda label: iris.target_names[label]
df['species'] = get_target_names(iris.target)

print(df)
print(df.info())
